\begin{frame}[fragile]
\begin{schemecode}
(define (factorial n)
  (cond (<= n 1)
        1
        (* n (factorial (- n 1)))))
\end{schemecode}
\end{frame}

(define (map proc items)

\end{schemecode}

(define (abs x)
  (cond ((> x 0) x)
        ((= x 0) x)

\begin{frame}[fragile]
Evaluation by substitution

\begin{schemecode}
(define (sum-of-squares x y)
  (+ (sqr x) (sqr y)))
\end{schemecode}

\nl
\hrule

     (sum-of-squares 3 4)
\end{schemecode}

\end{schemecode}
